Meters

Server meters.
class supriya.realtime.meters.Meters(server)[source]¶

Server meters.

>>> import json, supriya, time
>>> server = supriya.Server.default().boot(
...     input_bus_channel_count=2,
...     output_bus_channel_count=2,
...     )
>>> meters = server.meters.allocate()
>>> time.sleep(1)
>>> print(json.dumps(meters.to_dict(), indent=4))
{
    "server_meters": {
        "input_meter_levels": [
            {
                "peak": ...,
                "rms": ...
            },
            {
                "peak": ...,
                "rms": ...
            }
        ],
        "output_meter_levels": [
            {
                "peak": 0.0,
                "rms": 0.0
            },
            {
                "peak": 0.0,
                "rms": 0.0
            }
        ]
    }
}

>>> meters.free()
